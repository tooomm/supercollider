76  void open_client(std::string const & server_name, std::string const & name, uint32_t input_port_count,
77  uint32_t output_port_count, uint32_t blocksize)
78  {
79  blocksize_ = blocksize;
80 
81  /* open client */
82  client = server_name.empty() ? jack_client_open(name.c_str(), JackNoStartServer, &status)
83  : jack_client_open(name.c_str(), jack_options_t(JackNoStartServer | JackServerName),
84  &status, server_name.c_str());
85  std::atomic_thread_fence(std::memory_order_release); // ensure visibility on other threads
86 
87  if (status & JackServerFailed)
88  throw std::runtime_error("Unable to connect to JACK server");
89 
90  if (status & JackNameNotUnique) {
91  const char * client_name = jack_get_client_name(client);
92  std::cout << "unique client name: " << client_name << std::endl;
93  }
94 
95  /* initialize callbacks */
96  jack_set_thread_init_callback (client, jack_thread_init_callback, this);
97  jack_set_process_callback (client, jack_process_callback, this);
98  jack_set_xrun_callback(client, jack_xrun_callback, this);
99  jack_on_info_shutdown(client, (JackInfoShutdownCallback)jack_on_info_shutdown_callback, nullptr);
100 
101  /* register ports */
102  input_ports.clear();
103  for (uint32_t i = 0; i != input_port_count; ++i) {
104  std::string portname ("input_");
105  portname += std::to_string(i+1);
106  jack_port_t * port =
107  jack_port_register(client, portname.c_str(), JACK_DEFAULT_AUDIO_TYPE, JackPortIsInput, 0);
108  input_ports.push_back(port);
109 
110 #ifdef SC_JACK_USE_METADATA_API
111  jack_uuid_t uuid = jack_port_uuid(port);
112  if(!jack_uuid_empty(uuid)) {
113  std::string prettyname ("Input ");
114  prettyname += std::to_string(i+1);
115  jack_set_property(client, uuid,
116  JACK_METADATA_PRETTY_NAME, prettyname.c_str(), "text/plain");
117 
118  std::string order ("");
119  order += std::to_string(i);
120  jack_set_property(client, uuid,
121  JACKEY_ORDER, order.c_str(), "http://www.w3.org/2001/XMLSchema#integer");
122  }
123 #endif
124  }
125  input_channels = input_port_count;
126  super::input_samples.resize(input_port_count);
127 
128  output_ports.clear();
129  for (uint32_t i = 0; i != output_port_count; ++i) {
130  std::string portname ("output_");
131  portname += std::to_string(i+1);
132  jack_port_t * port =
133  jack_port_register(client, portname.c_str(), JACK_DEFAULT_AUDIO_TYPE, JackPortIsOutput, 0);
134  output_ports.push_back(port);
135 
136 #ifdef SC_JACK_USE_METADATA_API
137  jack_uuid_t uuid = jack_port_uuid(port);
138  if(!jack_uuid_empty(uuid)) {
139  std::string prettyname ("Output ");
140  prettyname += std::to_string(i+1);
141  jack_set_property(client, uuid, JACK_METADATA_PRETTY_NAME,
142  prettyname.c_str(), "text/plain");
143 
144  std::string order ("");
145  order += std::to_string(input_port_count + i);
146  jack_set_property(client, uuid, JACKEY_ORDER,
147  order.c_str(), "http://www.w3.org/2001/XMLSchema#integer");
148  }
149 #endif
150  }
151  output_channels = output_port_count;
152  super::output_samples.resize(output_port_count);
153 
154  samplerate_ = jack_get_sample_rate(client);
155  jack_frames = jack_get_buffer_size(client);
156 
157  if (jack_frames % blocksize_)
158  throw std::runtime_error("Jack buffer size is not a multiple of blocksize");
159 
160  cpu_time_accumulator.resize( samplerate_, jack_frames, 1 );
161  }

214  int connect_all_inputs(const char * client_name)
215  {
216  const char **ports = jack_get_ports (client, client_name, nullptr, JackPortIsOutput);
217 
218  if (!ports)
219  return -1;
220 
221  std::size_t i = 0;
222  while (ports[i]) {
223  if (i == input_ports.size())
224  break;
225 
226  int err = jack_connect(client, ports[i], jack_port_name(input_ports[i]));
227  if (err)
228  return err;
229  ++i;
230  }
231 
232  free(ports);
233  return 0;
234  }

303  int perform(jack_nframes_t frames)
304  {
305  if (unlikely(!time_is_synced)) {
306  engine_functor::sync_clock();
307  time_is_synced = true;
308  }
309 
310  /* get port regions */
311  jack_default_audio_sample_t ** inputs = (jack_default_audio_sample_t **)alloca(input_channels * sizeof(jack_default_audio_sample_t*));
312  jack_default_audio_sample_t ** outputs = (jack_default_audio_sample_t **)alloca(output_channels * sizeof(jack_default_audio_sample_t*));
313  for (uint16_t i = 0; i != input_channels; ++i)
314  inputs[i] = (jack_default_audio_sample_t*) jack_port_get_buffer(input_ports[i], frames);
315 
316  for (uint16_t i = 0; i != output_channels; ++i)
317  outputs[i] = (jack_default_audio_sample_t*) jack_port_get_buffer(output_ports[i], frames);
318 
319  jack_nframes_t processed = 0;
320  while (processed != frames) {
321  super::fetch_inputs((const float**)inputs, blocksize_, input_channels);
322  engine_functor::run_tick();
323  super::deliver_outputs(outputs, blocksize_, output_channels);
324  processed += blocksize_;
325  }
326 
327  cpu_time_accumulator.update(jack_cpu_load(client));
328 
329  return 0;
330  }

343  return 0;
344  }
