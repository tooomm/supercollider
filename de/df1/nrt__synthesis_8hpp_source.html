57  non_realtime_synthesis_engine(server_arguments const & args)

59  int format = headerFormatFromString(args.header_format.c_str())
60  | sampleFormatFromString(args.sample_format.c_str());

62  string input_file = args.input_file;
63  if (input_file == string("_"))

66  backend.open_client(input_file, args.output_file, args.samplerate, format, args.output_channels, args.blocksize);

70  has_inputs = !input_file.empty();
71  samples_per_block = args.blocksize;
72  prepare_backend(args.blocksize, args.input_channels, args.output_channels);

75  void prepare_backend(int blocksize, int input_channels, int output_channels)
76  {
77  std::vector<sample*> inputs, outputs;
78  for (int channel = 0; channel != input_channels; ++channel)
79  inputs.push_back(sc_factory->world.mAudioBus + (blocksize * (output_channels + channel)));
80 
81  backend.input_mapping(inputs.begin(), inputs.end());
82 
83  for (int channel = 0; channel != output_channels; ++channel)
84  outputs.push_back(sc_factory->world.mAudioBus + blocksize * channel);
85 
86  backend.output_mapping(outputs.begin(), outputs.end());
87  }

89  void run(void)
90  {
91  using namespace std;
92  using namespace std::chrono;
93 
94  const int reserved_packed_size = 16384;
95  std::vector <char> packet_vector(reserved_packed_size, 0);
96 
97  printf("\nPerforming non-rt synthesis:\n");
98  backend.activate_audio();
99 
100  auto start_time = steady_clock::now();
101 
102  while (!command_stream.eof()) {
103  boost::endian::big_int32_t packet_size;
104  command_stream.read((char*)&packet_size, sizeof(packet_size));
105 
106  assert(packet_size > 0);
107 
108  const bool huge_packet = (packet_size >= reserved_packed_size);
109 
110  if (huge_packet)
111  packet_vector.resize(packet_size);
112 
113  command_stream.read(packet_vector.data(), packet_size);
114 
115  time_tag bundle_time = instance->handle_bundle_nrt(packet_vector.data(), packet_size);
116 
117  size_t seconds = bundle_time.get_secs();
118  size_t nano_seconds = bundle_time.get_nanoseconds();
119  printf(" Next OSC bundle: %zu.%09zu\n", seconds, nano_seconds);
120 
121  while (instance->current_time() < bundle_time) {
122  if (instance->quit_requested())
123  goto done;
124 
125  if (has_inputs)
126  backend.audio_fn(samples_per_block);
127  else
128  backend.audio_fn_noinput(samples_per_block);
129  }
130 
131  if (huge_packet)
132  packet_vector.resize(16384);
133  }
134 
135  done:
136  backend.deactivate_audio();
137  auto end_time = steady_clock::now();
138  std::string elapsed_string = format_duration ( end_time - start_time );
139 
140  printf("\nNon-rt synthesis finished in %s\n", elapsed_string.c_str());
141 
142  auto peaks = backend.get_peaks();
143  printf("Peak summary:\n");
144  for (size_t channel = 0; channel != peaks.size(); ++channel) {
145  auto amplitude = peaks[channel];
146  printf(" Channel %zu: %gdB\n", channel, sc_ampdb(amplitude));
147  }
148  }

150  template <typename Duration>
151  static std::string format_duration(Duration const & duration)
152  {
153  using namespace boost;
154  using namespace std::chrono;
155 
156  auto elapsed_hours = duration_cast<hours>(duration);
157  auto remain = duration - elapsed_hours;
158  auto elapsed_minutes = duration_cast<minutes>(remain);
159  remain -= elapsed_minutes;
160  auto elapsed_nanoseconds = duration_cast<nanoseconds>(remain);
161  double seconds = elapsed_nanoseconds.count() * 1e-9;
162 
163  if (elapsed_hours.count())
164  return str( format("%|d|h %|d|m %|0.3f|s") % elapsed_hours.count() % elapsed_minutes.count() % seconds );
165  if (elapsed_minutes.count())
166  return str( format("%|d|m %|.3f|s") % elapsed_minutes.count() % seconds );
167  else
168  return str( format("%|.3f|s") % seconds );
169  }
