225  /* move node to head or tail of target */
226  template <node_position Position>
227  static void move_to_head_or_tail(server_node * node, abstract_group * target)
228  {
229  assert((Position == head) || (Position == tail));
230 
231  server_node_list::const_iterator target_iterator = (Position == head) ? target->child_nodes.begin()
232  : target->child_nodes.end();
233 
234  server_node_list::const_iterator node_iterator = server_node_list::s_iterator_to(*node);
235 
236  abstract_group * node_parent = node->get_parent();
237  if (node_parent != target) {
238  if (node->is_synth()) {
239  node_parent->child_synth_count -= 1;
240  target->child_synth_count += 1;
241  } else {
242  node_parent->child_group_count -= 1;
243  target->child_group_count += 1;
244 
245  group_list::const_iterator group_target_iterator = (Position == head) ? target->child_groups.begin()
246  : target->child_groups.end();
247 
248  abstract_group * node_as_group = static_cast<abstract_group*>(node);
249  target->child_groups.splice(group_target_iterator, node_parent->child_groups,
250  group_list::s_iterator_to(*node_as_group));
251  }
252  node->parent_ = target;
253  }
254 
255  target->child_nodes.splice(target_iterator, node_parent->child_nodes, node_iterator);
256  }

258  template <node_position Relation>
259  static void move_before_or_after(server_node * node, server_node * target)
260  {
261  assert((Relation == before) || (Relation == after));
262  abstract_group * target_parent = node->get_parent();
263 
264  if (Relation == after && target->next_node() == nullptr) {
265  // for the sake of simplicity, move the node to the tail of the target's parent group
266  move_to_head_or_tail<tail>(node, target_parent);
267  return;
268  }
269 
270  server_node * node_behind = (Relation == before) ? target
271  : target->next_node();
272  assert(node_behind);
273  server_node_list::const_iterator target_iterator = server_node_list::s_iterator_to(*node_behind);
274  server_node_list::const_iterator node_iterator = server_node_list::s_iterator_to(*node);
275 
276  abstract_group * node_parent = node->get_parent();
277  if (node_parent != target_parent) {
278  if (node->is_synth()) {
279  node_parent->child_synth_count -= 1;
280  target_parent->child_synth_count += 1;
281  } else {
282  node_parent->child_group_count -= 1;
283  target_parent->child_group_count += 1;
284 
285  group_list::const_iterator group_target_iterator = group_list::s_iterator_to(*static_cast<abstract_group*>(node));

293  target_parent->child_nodes.splice(target_iterator, node_parent->child_nodes, node_iterator);
