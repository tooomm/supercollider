45 QcSignalSpy( QObjectProxy *proxy, const char *sigName,
46  Qt::ConnectionType conType = Qt::QueuedConnection )
47 : QObject( proxy ), _proxy( proxy ), _sigId( -1 )
48 {
49 
50  Q_ASSERT( sigName );
51 
52  const QMetaObject *mo = _proxy->object()->metaObject();
53 
54  QByteArray signal = QMetaObject::normalizedSignature( sigName );
55  int sigId = mo->indexOfSignal( signal );
56 
57  if( sigId < 0 ) {
58  qcDebugMsg( 0, QString("WARNING: No such signal to connect: '%1'").arg(signal.constData()) );
59  return;
60  }
61 
62  int slotId = QObject::staticMetaObject.methodCount();
63 
64  if( !QMetaObject::connect( _proxy->object(), sigId, this, slotId,
65  conType, 0) )
66  {
67  qcErrorMsg( "QMetaObject::connect returned false. Unable to connect." );
68  return;
69  }
70 
71  QMetaMethod mm = mo->method( sigId );
72 
73  QList<QByteArray> params = mm.parameterTypes();
74 
75  for( int i = 0; i < params.count(); ++i ) {
76  int type = QMetaType::type( params.at(i).constData() );
77  if( type == QMetaType::Void )
78  qcErrorMsg( QString("QObject:connect: Don't know how to handle '%1', "
79  "use qRegisterMetaType to register it.")
80  .arg(params.at(i).constData()) );
81  _argTypes << type;
82  }
83 
84  _sigId = sigId;
85 }

87 int qt_metacall( QMetaObject::Call call, int methodId, void **argData )
88 {
89  methodId = QObject::qt_metacall( call, methodId, argData );
90 
91  if( methodId < 0 )
92  return methodId;
93 
94  if( call == QMetaObject::InvokeMetaMethod ) {
95  Q_ASSERT( methodId == 0 );
96 
97  QList<QVariant> args;
98  args.reserve( _argTypes.count() );
99 
100  for (int i = 0; i < _argTypes.count(); ++i) {
101  QMetaType::Type type = static_cast<QMetaType::Type>(_argTypes.at(i));
102  args << QVariant( type, argData[i + 1] );
103  }
104 
105  react( args );
106 
107  methodId = -1;
108  }
109 
110  return methodId;
111 }
