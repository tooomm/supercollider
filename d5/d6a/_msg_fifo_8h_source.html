32 using std::memory_order_relaxed;
33 using std::memory_order_release;

35 template <class MsgType, int N>
36 class MsgFifo
37 {
38 public:
39  MsgFifo()
40  : mReadHead(0), mWriteHead(0), mFreeHead(0)
41  {}
42 
43  void MakeEmpty() {
44  int writeHead= mWriteHead.load();
45  mReadHead.store(writeHead, memory_order_relaxed);
46  mFreeHead.store(writeHead, memory_order_release);
47  }
48  bool IsEmpty() { return mReadHead.load(memory_order_relaxed) == mWriteHead.load(memory_order_relaxed); }
49  bool HasData() { return mReadHead.load(memory_order_relaxed) != mWriteHead.load(memory_order_relaxed); }
50  bool NeedsFree() { return mFreeHead.load(memory_order_relaxed) != mReadHead.load(memory_order_relaxed); }
51 
52  bool Write(MsgType& data)
53  {
54  unsigned int next = NextPos(mWriteHead.load(memory_order_relaxed));
55  if (next == mFreeHead.load(memory_order_relaxed)) return false; // fifo is full
56  mItems[next] = data;
57 
58  mWriteHead.store(next, memory_order_release);
59 
60  return true;
61  }
62 
63  void Perform() // get next and advance
64  {
65  while (HasData()) {
66  unsigned int next = NextPos(mReadHead.load(memory_order_relaxed));
67  mItems[next].Perform();
68  mReadHead.store(next, memory_order_release);
69  }
70  }
71  void Free() // reclaim messages
72  {
73  while (NeedsFree()) {
74  unsigned int next = NextPos(mFreeHead.load(memory_order_relaxed));
75  mItems[next].Free();
76  mFreeHead.store(next, memory_order_release);
77  }
78  }
79 
80 private:
81  int NextPos(int inPos) { return (inPos + 1) & (N - 1); }
82 
83  std::atomic<int> mReadHead, mWriteHead, mFreeHead;
84  MsgType mItems[N];
85 };
