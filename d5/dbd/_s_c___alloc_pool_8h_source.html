198  static size_t BinIndex(size_t inSize)
199  {
200  if (inSize < 1024) return inSize >> 4;

222  int NextFullBin(int inStartingBinIndex)
223  {
224  if (inStartingBinIndex >= 128) return -1;
225  int word = inStartingBinIndex >> 5;
226  int bitPosition = inStartingBinIndex & 31;
227  unsigned long bitValue = 1L << bitPosition;
228  unsigned long binBits = mBinBlocks[word];
229 
230  if (binBits >= bitValue) {
231  binBits = (~(bitValue - 1) & binBits);
232  } else {
233  for (++word; word<4 && !mBinBlocks[word]; ++word) {}
234  if (word==4) return -1;
235  binBits = mBinBlocks[word];
236  }
237  bitPosition = CTZ((int32)binBits);
238 
239  return (word << 5) + bitPosition;
240  }
