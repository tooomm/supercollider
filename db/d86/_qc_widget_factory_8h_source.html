36  virtual QObjectProxy *newInstance( PyrObject *scObject, QtCollider::MetaValue arg[10] ) {
37 
38  // Omit the first two arguments - parent and bounds
39 
40  QWIDGET *w;
41 
42  if( !arg[2].type() ) {
43  w = new QWIDGET;
44  }
45  else {
46  QObject *obj = QWIDGET::staticMetaObject.newInstance(
47  arg[2].toGenericArgument(),
48  arg[3].toGenericArgument(),
49  arg[4].toGenericArgument(),
50  arg[5].toGenericArgument(),
51  arg[6].toGenericArgument(),
52  arg[7].toGenericArgument(),
53  arg[8].toGenericArgument(),
54  arg[9].toGenericArgument()
55  );
56 
57  w = qobject_cast<QWIDGET*>(obj);
58  if( !w ) {
59  qcNoConstructorMsg( QcObjectFactory<QWIDGET>::metaObject(), 8, &arg[2] );
60  return 0;
61  }
62  }
63 
64  // NOTE: performance: it is completely equal if parent is passed
65  // in constructor, or set later, but for some reason it is cheaper
66  // if it is set here, before setting other stuff like geometry, etc.
67 
68  QObjectProxy *parentProxy( arg[0].value<QObjectProxy*>() );
69  QWidget *parent = parentProxy ? qobject_cast<QWidget*>( parentProxy->object() ) : 0;
70 
71  if( parent )
72  {
73  const QMetaObject *mo = parent->metaObject();
74  int mi = mo->indexOfMethod( "addChild(QWidget*)" );
75  bool ok;
76  if( mi >= 0 )
77  {
78  QMetaMethod mm = mo->method( mi );
79  ok = mm.invoke( parent, Q_ARG(QWidget*, w) );
80  if(!ok)
81  {
82  qcErrorMsg("Could not set parent!");
83  delete w;
84  return 0;
85  }
86  }
87  else
88  {
89  if(parent->layout())
90  parent->layout()->addWidget(w);
91  else
92  w->setParent( parent );
93  }
94  }
95 
96  // set requested geometry
97 
98  QRect r( arg[1].value<QRectF>().toRect() );
99  if( r.size().isEmpty() ) r.setSize( w->sizeHint() );
100 
101  w->setGeometry( r );
102 
103  // automatically support drag and drop
104 
105  w->setAcceptDrops( true );
106 
107  // ensure visible:
108 
109  if(parent && parent->isVisible()) w->show();
110 
111  // create the proxy:
112 
113  QObjectProxy *prox( proxy(w, scObject) );
114 
115  return prox;
116  }
117 
118 protected:
119

121  {
122  QWidgetProxy *prox( new QWidgetProxy( w, sc_obj ) );
123  initialize( prox, w );
124  return prox;
125  }

128 };
