83  case after: {

97  abstract_group * group = static_cast<abstract_group*>(reference_node);
98  group->add_child(node);
99  break;
100  }

104  if (reference_node_is_group) {
105  abstract_group * reference_group = static_cast<abstract_group*>(reference_node);
106  reference_group->apply_deep_on_children( [&] (server_node & node) {
107  doOnFree(node);
108  });
109  }
110  doOnFree(*reference_node);
111 
112  abstract_group * node_parent = reference_node->parent_;
113  node_parent->replace_child(node, reference_node);
114 
115  if (reference_node_is_group)
116  group_count_ -= 1;
117  else

119 
120  break;
121  }
122 
123  default:
124  assert(false); /* this point should not be reached! */
