46 class server_shared_memory
47 {
48 public:
49  typedef offset_ptr<float> sh_float_ptr;
50  typedef offset_ptr<scope_buffer> scope_buffer_ptr;
51 
52  typedef bi::allocator<scope_buffer_ptr, managed_shared_memory::segment_manager> scope_buffer_ptr_allocator;
53  typedef bi::vector<scope_buffer_ptr, scope_buffer_ptr_allocator> scope_buffer_vector;
54 
55  server_shared_memory(managed_shared_memory & segment, int control_busses, int num_scope_buffers = 128):
56  num_control_busses(control_busses),
57 
58  scope_buffers(scope_buffer_ptr_allocator(segment.get_segment_manager()))
59  {
60  control_busses_ = (float*)segment.allocate(control_busses * sizeof(float));
61  std::fill(control_busses_.get(), control_busses_.get() + control_busses, 0.f);
62 
63  for (int i = 0; i != num_scope_buffers; ++i) {
64  scope_buffer * raw_scope_ptr = (scope_buffer*)segment.allocate(sizeof(scope_buffer));
65  new(raw_scope_ptr) scope_buffer();
66  scope_buffer_ptr buf = raw_scope_ptr;
67  scope_buffers.push_back(buf);
68  }
69  }
70 
71  void destroy(managed_shared_memory & segment)
72  {
73  segment.deallocate(control_busses_.get());
74 
75  for (size_t i = 0; i != scope_buffers.size(); ++i)
76  segment.deallocate(scope_buffers[i].get());
77  }

104 class server_shared_memory_creator
105 {
106 public:
107  server_shared_memory_creator(unsigned int port_number, unsigned int control_busses):
108  shmem_name(detail_server_shm::make_shmem_name(port_number)),
109  segment(bi::open_or_create, shmem_name.c_str(), 8192 * 1024)
110  {
111 #if (BOOST_VERSION < 105100)
112  segment.flush();
113 #endif
114 
115  const int num_scope_buffers = 128;
116  size_t scope_pool_size = num_scope_buffers * sizeof(float) * 8192; // pessimize, about 4 MB
117  void * memory_for_scope_pool = segment.allocate(scope_pool_size);
118  scope_pool.init(memory_for_scope_pool, scope_pool_size);
119 
120  shm = segment.construct<server_shared_memory>(shmem_name.c_str())(ref(segment), control_busses,
121  num_scope_buffers);
122  }
