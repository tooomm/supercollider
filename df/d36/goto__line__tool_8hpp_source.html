110  void go()
111  {
112  int lineNumber = mSpinBox->value();
113 
114  if (mEditor) {
115  QTextDocument *doc = mEditor->textDocument();
116  QTextBlock block( doc->findBlockByNumber(lineNumber - 1) );
117  if (!block.isValid())
118  return;
119 
120  QTextCursor cursor( doc );
121  cursor.setPosition( block.position() );
122  mEditor->setTextCursor(cursor);
123  }
124 
125  emit activated( lineNumber );
126  }
