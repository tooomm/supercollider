54  template <typename reverse_iterator>
55  HOT static inline int get_previous_activation_count(reverse_iterator it, reverse_iterator end, int previous_activation_limit)
56  {
57  reverse_iterator prev = it;
58 
59  for(;;) {
60  ++prev;
61  if (prev == end)
62  return previous_activation_limit; // we are the first item, so we use the previous activiation limit
63 
64  server_node & node = *prev;
65  if (node.is_synth())
66  return 1;
67  else {
68  abstract_group & grp = static_cast<abstract_group&>(node);
69  int tail_nodes = grp.tail_nodes();
70 
71  if (tail_nodes != 0) /* use tail nodes of previous group */
72  return tail_nodes;
73  else /* skip empty groups */
74  continue;
75  }
76  }

87  HOT successor_container fill_queue_recursive(group & g, successor_container const & successors_from_parent, size_t previous_activation_limit)
88  {
89  assert (g.has_synth_children());
90 
91  typedef server_node_list::reverse_iterator r_iterator;
92 
93  successor_container successors(successors_from_parent);
94 
95  size_t children = g.child_count();
96 
97  sequential_child_list sequential_children;
98  sequential_children.reserve(g.child_synth_count);
99 
100  for (r_iterator it = g.child_nodes.rbegin(); it != g.child_nodes.rend(); ++it) {
101  server_node & node = *it;
102 
103  if (node.is_synth()) {
104  r_iterator end_of_node = it;
105  --end_of_node; // one element behind the last
106  std::size_t node_count = 1;
107 
108  // we fill the child nodes in reverse order to an array
109  for(;;) {
110  sequential_children.push_back(&*it);
111  ++it;
112  if (it == g.child_nodes.rend())
113  break; // we found the beginning of this group
114 
115  if (!it->is_synth())
116  break; // we hit a child group, later we may want to add it's nodes, too?
117  ++node_count;
118  }
119 
120  --it; // we iterated one element too far, so we need to go back to the previous element
121  assert(sequential_children.size() == node_count);
122 
123  auto seq_it = sequential_children.rbegin();
124 
125  int activation_limit = get_previous_activation_count(it, g.child_nodes.rend(), previous_activation_limit);
126 
127  thread_queue_item * q_item =
128  q->allocate_queue_item(queue_node(std::move(queue_node_data(static_cast<abstract_synth*>(*seq_it++))), node_count),
129  successors, activation_limit);
130 
131  queue_node & q_node = q_item->get_job();
132 
133  // now we can add all nodes sequentially
134  for(;seq_it != sequential_children.rend(); ++seq_it)
135  q_node.add_node(static_cast<abstract_synth*>(*seq_it));
136  sequential_children.clear();
137 
138  assert(q_node.size() == node_count);
139 
140  /* advance successor list */
141  successors = successor_container(1);
142  successors[0] = q_item;
143 
144  if (activation_limit == 0)
145  q->add_initially_runnable(q_item);
146  children -= node_count;
147  } else {
148  abstract_group & grp = static_cast<abstract_group&>(node);
149 
150  if (grp.has_synth_children()) {
151  int activation_limit = get_previous_activation_count(it, g.child_nodes.rend(), previous_activation_limit);
152  successors = fill_queue_recursive(grp, successors, activation_limit);
153  }
154 
155  children -= 1;
156  }
157  }
158  assert(children == 0);
159  return successors;
160  }

162  HOT successor_container fill_queue_recursive(parallel_group & g, successor_container const & successors_from_parent, size_t previous_activation_limit)
163  {
164  assert (g.has_synth_children());
165  std::vector<thread_queue_item*, rt_pool_allocator<thread_queue_item*> > collected_nodes;
166  collected_nodes.reserve(g.child_synth_count + g.child_group_count * 16); // pessimize
167 
168  for (auto & node : g.child_nodes) {
169 
170 
171  if (node.is_synth()) {
172  thread_queue_item * q_item = q->allocate_queue_item(queue_node(std::move(queue_node_data(static_cast<abstract_synth*>(&node)))),
173  successors_from_parent, previous_activation_limit);
174 
175  if (previous_activation_limit == 0)
176  q->add_initially_runnable(q_item);
177 
178  collected_nodes.push_back(q_item);
179  } else {
180  abstract_group & grp = static_cast<abstract_group&>(node);
181 
182  if (grp.has_synth_children()) {
183  successor_container group_successors = fill_queue_recursive(grp, successors_from_parent,
184  previous_activation_limit);
185 
186  for (unsigned int i = 0; i != group_successors.size(); ++i)
187  collected_nodes.push_back(group_successors[i]);
188  }
189  }
190  }
191 
192  successor_container ret(collected_nodes.size());
193 
194  memcpy(ret.data->content, collected_nodes.data(), collected_nodes.size() * sizeof(thread_queue_item *));
195  return ret;
196  }
