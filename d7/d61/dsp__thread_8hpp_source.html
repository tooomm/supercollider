99  void start()
100  {
101  stop = false;
102 
103 #ifdef SUPERNOVA_USE_PTHREAD
104  pthread_attr_t attr;
105  pthread_attr_init(&attr);
106  int err = pthread_attr_setstack(&attr, stack_, stack_size);
107  if (err)
108  throw std::logic_error("Cannot set stack of DSP helper thread");
109 
110  err = pthread_create( &thread_id, &attr, run_static, this );
111  if (err)
112  throw std::runtime_error("Cannot create DSP helper thread");
113  pthread_attr_destroy(&attr);
114 #else
115  thread = std::thread([this] {this->run();});
116 #endif
117  }

119  void join()
120  {
121  stop.store( true, std::memory_order_relaxed );
122  wake_thread();
123 
124 #ifdef SUPERNOVA_USE_PTHREAD
125  void * ret;
126  int err = pthread_join( thread_id, &ret );
127  if (err)
128  printf("Error when joining helper thread\n");
129 #else
130  thread.join();
131 #endif
132  }
