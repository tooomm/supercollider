103  buffer_lock2(const SndBuf * buf1, const SndBuf * buf2):
104  buf1_(buf1), buf2_(buf2)
105  {
106  if (buf1 == buf2) {
107  lock1();
108  return;
109  }
110 
111  for(;;) {
112  lock1();
113 
114  if (lock2())
115  return;
116  unlock1();
117  }
118  }

121  {
122  unlock1();
123  if (buf1_ != buf2_)
124  unlock2();
125  }

129  {
